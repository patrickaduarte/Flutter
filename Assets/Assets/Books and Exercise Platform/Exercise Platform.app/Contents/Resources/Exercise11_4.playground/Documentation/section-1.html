var fullNames: [String] = []

for person in people {
    var fullName = " "
    for (key, value) in person {
        if key == "lastName" {
            fullName += value
        } else {
            fullName = value + fullName
        }
    }
    fullNames += [fullName]
}

print(fullNames)
