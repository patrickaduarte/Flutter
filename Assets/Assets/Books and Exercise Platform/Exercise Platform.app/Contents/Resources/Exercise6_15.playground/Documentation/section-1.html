var listOfNumbers = [1, 2, 3, 1, 2, 10, 100]

var unique: [Int] = []

for number in listOfNumbers {
    var numberIsNew = true

    for otherNumber in unique {
        if number == otherNumber {
            numberIsNew = false
            break
        }
    }

    if numberIsNew {
        unique.append(number)
    }
}

for number in unique {
    print(number)
}

We'll initialize our unique array to be empty. Next we iterate through all the numberes in listOfNumbers for each number we check if its already in our unique array, if it is not we add it to the unique array. This will result in an array with all the unique numbers from listOfNumbers.
